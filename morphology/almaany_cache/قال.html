    • قَالَ قَوْلاً، ومَقالاً، ومقالة: ق تكلّم تَلَفَّظَ، خَاطَبَ ،
    • قال فلانًا صادقًا: ظنَّه صادقًا وقال هنا تنصب مفعولين
    • قال في نفسه كذا: حدثته نفسه به
    • قال بأنه سيأتي غدًا/ قال عنه إنه سيأتي غدًا: أخبر، روى

    • قال في الموضوع برأيه: اجتهد

  2. قالَ: (فعل)
    • قالَ يَقِيل ، قِلْ ، قَيْلاً وقَيْلولةً ، فهو قائلٌ والجمع : قُيّلٌ، وقُيَّالٌ
    • قال الشَّخصُ: نام في منتصف النّهار
    • قَالَ شَرِب اللَّبَنَ في القائلة
    • قَالَ فلانا البَيْعَ: فَسَخَهُ
  3. قَالٍ: (اسم)
    • قَالٍ : فاعل من قليَ
  4. قال: (اسم)
    • مصدر قَالَ
    • القَالُ : اسمٌ من القول
    • الْقَالُ وَالْقِيلُ : مَا يَقُولُهُ النَّاسُ عَنْ بَعْضِهِمْ ، إشاعات كاذبة ، ثرثرة، هذر وتخليط، فضول الكلام أو الغيبة والنميمة ، ونهى النبي صلى الله عليه وسلم عن القِيل والقال/ اي عن فضول القول مما يوقع الخصومة بين الناس
    • ورجل قال: قائلٌ
    • القَالُ :القُلَة، أَو خَشَبتها التي تُضْرَب بها والجمع : هذه: قيلان
  5. قال: (اسم)
    • قال : فاعل من قالَ
  6. قالٍ: (اسم)
    • قالٍ : فاعل من قلَى
  7. قالّ: (اسم)
    • قالّ : فاعل من قلَّ
  8. قَوَل: (اسم)
    • قَوَل : جمع قَوَّالُ
  9. قَوَّلَ: (فعل)
    • قوَّلَ يقوِّل ، تقويلاً ، فهو مُقَوِّل ، والمفعول مُقَوَّل
    • قوَّله الحقَّ :جعله يَقُولُه
    • قَوَّلَهُ شِعْراً : حَفَّظَهُ، عَلَّمَهُ إِيَّاهُ
    • قوَّله ما لم يقُل: ادَّعاه عليه
  10. قَول: (اسم)
    • الجمع : أقوال ، أقاويلُ
    • مصدر قالَ
    • قَول: كلام : أقوال كاذبة مفتراة
    • ابن أقوال: بمعنى الإصابة في المنطق والقدرة على الكلام؛ ويشير إلى معاني حسن الكلام والأداء اللغوي،
    • القول السَّديد: كلام يتميَّز بإصابة المعنى إصابة كاملة بطريقة فيها الإيجاز البليغ، وجذب الانتباه، والمطابقة لمقتضى الحال. والقول السديد ما كان صواباً تسد به الحاجة هو كل كلمة تسد منافذ الشيطان وكذلك منافذ الشك والريبة، فكل كلمة طيبة هي قول سديد
    • قولاً وعملاً: حقيقةً يعمل ما يقول،
    • قوْل فَصْل: قول حقّ ليس بباطل،
    • قول مأثور: حكمة تداولها الناس، وتميّزت بالدّلالة مع الإيجاز،
    • مجرد أقاويل: مجرد شائعات أو كلام غير مؤكّد
    • قَول: رأي، مذهب
    • قَول: شهادة أو إقرار إطلاقًا أعطى قَوْلَه في القضيَّة،
    • أعطى قوْلاً: ارتبط بكلامه، التزم بوعده
    • قَول: طَلَب
    • إِنْ صَحَّ القَوْلُ : إِنْ صَحَّ التَّعْبِيرُ
    • إِنَّهُ القَوْلُ الفَصْلُ : الْحُكْمُ الفَاصِلُ بَيْنَ الْحَقِّ وَالبَاطِلِ
  11. قُوُل: (اسم)
    • قُوُل : جمع قَوُولُ
  12. قُوْل: (اسم)
    • قُوْل : جمع قَؤُولُ
  13. قُوَّل: (اسم)
    • قُوَّل : جمع قائِلُ
  14. قُول: (اسم)
    • قُول : جمع قُوَلة
  15. قول: (اسم)
    • الجمع : أقْوال ، و الجمع : أَقاويل
    • القوْلُ : الكلامُ
    • القوْلُ :الرأيُ والمعتقَد
  16. أَقاويل: (اسم)
    • أَقاويل : جمع قول
  17. تَقَوَّلَ: (فعل)
    • تقوَّلَ يتقوَّل ، تقوُّلاً ، فهو مُتَقَوِّل ، والمفعول مُتَقَوَّل
    • تَقَوَّلَ عَلَيْهِ : قالَ عَنْهُ كَلاماً فيهِ افْتِراءٌ وَكَذِبٌ
  18. قاوَلَ: (فعل)
    • قاولَ يُقاول ، مُقاولَةً ، فهو مُقاوِل ، والمفعول مُقاوَل
    • قَاوَلَ صَاحِبَهُ : بَاحَثَهُ، جَادَلَهُ
    • قَاولَهُ في الأمر: فاوضَه وجادله
    • قاول بنَّاءً: أعطاه العمل مقاولةً على تعهّد منه بالقيام به
  19. إِقْوَال: (اسم)
    • إِقْوَال : مصدر أَقْوَلَ

تعريف و معنى في قاموس الكل. قاموس عربي عربي

  1. قال بأنه سيأتي غدا/ قال عنه إنه سيأتي غدا
    • أخبر، روى.

    المعجم: عربي عامة

  2. قالَ
    • قالَ / قالَ بـ / قالَ عن / قالَ في / قالَ لـ يقول ، قُلْ ، قولاً وقالاً وقِيلاً وقَالةً ، فهو قائل ، والمفعول مقول :-
      • قال فلانٌ رأيَه تحدَّث به، تكلّم به، لفظه :-قال إنّه متضرِّر ممّا يحدث، - قال متمتمًا/ متلعثمًا/ متلجلجًا، - *فقالت له العينان سمعًا وطاعة*، - فإن لم تجد قولاً سديدًا تقوله ... فصمتك عن غير السديد سدادُ، - {وَإِذْ قَالَ مُوسَى لِقَوْمِهِ يَاقَوْمِ إِنَّكُمْ ظَلَمْتُمْ أَنْفُسَكُمْ} - {وَإِذَا قُلْتُمْ فَاعْدِلُوا}: إذا شهدتم، - {وَمَنْ أَصْدَقُ مِنَ اللهِ قِيلاً} :-
      • قال كلامًا من ذهب: تفوَّه بأحسن وأفضل ما يكون من الكلام.
      • قالَ عليه كلامًا زورًا: افترى، ادّعى :- {كَبُرَ مَقْتًا عِنْدَ اللهِ أَنْ تَقُولُوا مَا لاَ تَفْعَلُونَ} - {يَاشُعَيْبُ مَا نَفْقَهُ كَثِيرًا مِمَّا تَقُولُ} .
      • قال فلانًا صادقًا؟: ظنَّه صادقًا وقال هنا تنصب مفعولين.
      • قال في نفسه كذا: حدثته نفسه به.
      • قال بأنه سيأتي غدًا/ قال عنه إنه سيأتي غدًا: أخبر، روى.
      • قال بآراء السلف: رآه رأيًا واعتقده.
      • قال بيده: أشار بها :-قال برأسه.
      • قال في الموضوع برأيه: اجتهد.
      • قال لصديقه إنّك ناجح: خاطبه :-قلت له أن يفعل ذلك.

    المعجم: اللغة العربية المعاصر

  3. القَالُ
    • القَالُ : اسمٌ من القول.
      ونهى النبي صلى الله عليه وسلم عن القِيل والقال؛ ويُروى: حديث شريف عن قِيلَ وقالَ // بلفظ الفعل الماضي: عن فضول القول مما يوقع الخصومة بين الناس.
      ورجل قال: قائلٌ.
      و القَالُ القُلَة، أَو خَشَبتها التي تُضْرَب بها. والجمع : هذه: قيلان.
      (انظر: قلا) .

    المعجم: المعجم الوسيط

  4. قال الشّخص
    • نام في منتصف النّهار :- {فَجَاءَهَا بَأْسُنَا بَيَاتًا أَوْ هُمْ قَائِلُونَ}.

    المعجم: عربي عامة

  5. قال بآراء السلف
    • رآه رأيًا واعتقده.

    المعجم: عربي عامة

  6. قال بيده
    • أشار بها :-قال برأسه.

    المعجم: عربي عامة

  7. قال عليه كلاما زورا
    • افترى، ادّعى :- {كَبُرَ مَقْتًا عِنْدَ اللهِ أَنْ تَقُولُوا مَا لاَ تَفْعَلُونَ} - {يَاشُعَيْبُ مَا نَفْقَهُ كَثِيرًا مِمَّا تَقُولُ}.

    المعجم: عربي عامة

    • تحدَّث به، تكلّم به، لفظه :-قال إنّه متضرِّر ممّا يحدث- قال متمتمًا/ متلعثمًا/ متلجلجًا- *فقالت له العينان سمعًا وطاعة*- فإن لم تجد قولاً سديدًا تقوله ... فصمتك عن غير السديد سدادُ- {وَإِذْ قَالَ مُوسَى لِقَوْمِهِ يَاقَوْمِ إِنَّكُمْ ظَلَمْتُمْ أَنْفُسَكُمْ} - {وَإِذَا قُلْتُمْ فَاعْدِلُوا}

    المعجم: عربي عامة

  9. قال فلانا صادقا؟

  13. قَالَ
    • قَالَ قَالَ ُ قَوْلاً، ومَقالاً، ومقالة: قَالَ تكلّم.
      فهو قائِل، وقالٌ.
      وجمع قائل: قالَةٌ.
      ويُستعمَلُ القولُ مجازًا للدلالة على الحال مثل:

    المعجم: المعجم الوسيط

  14. قَالَ
    • قَالَ قَالَ ِ قَيْلاً: نام وَسطَ النهار.
      قَالَ فهو قائلٌ. والجمع : قُيّلٌ، وقُيَّالٌ.
      و قَالَ شَرِب اللَّبَنَ في القائلة.
      و قَالَ فلانا البَيْعَ: فَسَخَهُ.

  15. قَالَ
    • [ق و ل]. (فعل: ثلاثي لازم متعد بحرف). قُلْتُ، أَقُولُ، قُلْ، مصدر قَوْلٌ، قَالٌ، قَوْلَةٌ، مَقَالٌ، مَقَالَةٌ.
      1. :-قَالَ الرَّجُلُ :- : تَلَفَّظَ، خَاطَبَ. :-قَالَ لَهُ مُبْتَسِماً : تَفَضَّلْ.
      2. :-يَقُولُ مَا لاَ يَفْعَلُ :- : يَفُوهُ.
      3. :-قَالَ عَلَيْهِ :- : اِفْتَرَى عَلَيْهِ.
      4. :-قَالَ عَنْهُ :-: رَوَى عَنْهُ، أَخْبَرَ.
      5. :-قَالَ بِآرَائِهِ :- : اِعْتَقَدَ بِهَا.
      6. :-هَذَا مَا يُقَالُ :-: هَذَا مَا يُرْوَى وَيُحْكَى.
      7. :-قَالَ بِرَأْسِهِ :- : أَشَارَ بِهِ.
      8. :-قَالَ بِيَدِهِ :- : أَهْوَى بِهَا وَأَخَذَ.
      9. :-قَالَ فِيهِ :- : اِجْتَهَدَ. 10. :-قَالَ فَتَكَلَّمَ :- : يُعَبَّرُ بِفِعْلِ :-قَالَ :- هَذَا عَنِ التَّهَيُّؤِ لِلأَفْعَالِ وَالاسْتِعْدَادِ لَهَا. :-قَالَ فَذَهَبَ. 11. : يَأْتِي فِعْلُ :-قَالَ :- بِمَعْنَى :-ظَنَّ :- وَفِي هَذِهِ الْحَالَةِ يَنْصِبُ مَفْعُولَيْنِ بِشَرْطِ أَنْ يَكُونَ للاسْتِقْبَالِ عِنْدَ مُخَاطَبَةِ الْمُخَاطَبِ وَمَسْبُوقاً بِالاسْتِفْهَامِ وَلَمْ يَنْفَصِلْ عَنْهُ بِغَيْرِ ظَرْفٍ أَوْ جَارٍ وَمَجْرُورٍ أَوْ مَعْمُولِهِ: :-أَتَقُولُ الأُسْتَاذَ غَائِباً اليَوْمَ :- :-أَتَقُولُ نَهَاراً مُشْمِساً. 12. :-وَتُحْكَى الْجُمَلُ بَعْدَ القَوْلِ فَتُعَادُ بِلَفْظِهَا :مريم آية 30 قَالَ إِنِّيَ عَبْدُ اللَّهِ ( قرآن). أَوْ بِمَعْنَاهَا مِثْل : :-قَالَ إِنَّهُ عَبْدُ اللَّهِ. إذا فُصِلَ بِغَيْرِ مَا اشْتَرَطَ بَطُلَتْ مُوَافَقَةُ الظَّنّ وَ نُفِيَتِ الْحِكَايَةُ : :-أَأَنْتَ تَقُولُ زَيْدٌ رَاحِلٌ.

    المعجم: الغني

    المعجم: الغني

    المعجم: الغني

      2 - إشاعات كاذبة :-راج حول فلان بعض القيل والقال، - نَهَى النَّبِيّ صَلَّى اللهُ عَلَيْهِ وَسَلَّم عَنِ القِيلِ وَالقَالِ [حديث] :-
      • قالٌ وقِيل: ثرثرة، هذر وتخليط، فضول الكلام أو الغيبة والنميمة.
      3 - كلمة :- {ذَلِكَ عِيسَى ابْنُ مَرْيَمَ قَالُ الْحَقِّ الَّذِي فِيهِ يَمْتَرُونَ} [قرآن] .

أمثلة سياقية: قال، جمل ورد بها قال
يَا بَشِيْرُ ، أَلَكَ ابْنٌ غَيْرُ هَذَا ؟ '' قَالَ : نَعَمْ ، قَالَ : قَالَ : '' فَوَهَبْتَ لَهُ مِثْلَ الَّذِي وَهَبْتَ لِهَذَا ؟ '' قَالَ : لَا ، قَالَ : '' فَلَا تُشْهِدْنِي إِذًا ، فَإِنِّي لَا أَشْهَدُ عَلَى جَوْرٍ (حديث )
أَلَا أَدُلُّكَ عَلَى كَنْزٍ مِنْ كُنُوزِ الْجَنَّةِ '' , قُلْتُ : بَلَى , قَالَ : '' لَا حَوْلَ وَلَا قُوَّةَ إِلَّا بِاللَّهِ '' , قَالَ : أَحْسِبُهُ قَالَ : يَقُولُ اللَّهُ عَزَّ وَجَلَّ : '' أَسْلَمَ عَبْدِي وَاسْتَسْلَمَ (حديث )
لَوْلَا أَنَّ قَوْمَكِ حَدِيثٌ عَهْدُهُمْ بِجَاهِلِيَّةٍ ، أَوْ قَالَ : '' بِكُفْرٍ '' ، قَالَ : يَقُولُ ابْنُ الزُّبَيْرِ : '' لَنَقَضْتُ الْكَعْبَةَ ، فَجَعَلْتُ لَهَا بَابَيْنِ فِي الْأَرْضِ ، بَابًا يُدْخَلُ مِنْهُ ، وَبَابًا يُخْرَجُ مِنْهُ '' قَالَ أَبُو إِسْحَاقَ : فَأَنَا رَأَيْتُهَا كَذَلِكَ . (حديث )
قَالَ: ثُمَّ ذَكَرْتُ الْكَلِمَاتِ فَقُلْتُهُنَّ قَالَ: فَأَنِسْتُ قَلِيلًا وَرَجَعَتْ إِلَيَّ نَفْسِي (عامة )
أَلَا أَرْقِيكَ بِرُقْيَةِ رَسُولِ اللَّهِ ؟ قَالَ : بَلَى ، قَالَ : فَقَالَ : اللَّهُمَّ رَبَّ النَّاسِ ، مُذْهِبَ الْبَاسِ ، اشْفِ أَنْتَ الشَّافِي ، لَا شَافِيَ إِلَّا أَنْتَ اشْفِهِ شِفَاءً لَا يُغَادِرُ سَقَمًا (حديث )
